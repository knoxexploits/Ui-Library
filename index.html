Full Code Sample

This is a complete script showing how to build a real hub. It includes variable-based logic for speed and a clean layout.

local RedGlass = loadstring(game:HttpGet("https://raw.githubusercontent.com/knoxexploits/Ui-Library/refs/heads/main/RedGalss"))()

local Window = RedGlass.new({
    Name = "RedGlass Demo",
    ConfigurationSaving = true
})

-- Script Variables
local LP = game:GetService("Players").LocalPlayer
local SpeedVal = 16

-- Tabs
local Main = Window:CreateTab("Main")
local Misc = Window:CreateTab("Misc")

-- Main Tab Components
Main:CreateSection("Character Tweaks")

Main:CreateToggle({
    Name = "Enable Speed",
    CurrentValue = false,
    Flag = "speed_enabled",
    Callback = function(v)
        _G.SpeedEnabled = v
        while _G.SpeedEnabled do task.wait()
            if LP.Character and LP.Character:FindFirstChild("Humanoid") then
                LP.Character.Humanoid.WalkSpeed = SpeedVal
            end
        end
    end
})

Main:CreateSlider({
    Name = "Speed Amount",
    Min = 16, Max = 250, CurrentValue = 16,
    Flag = "speed_val",
    Callback = function(v)
        SpeedVal = v
    end
})

-- Misc Tab Components
Misc:CreateSection("Teleportation")

Misc:CreateDropdown({
    Name = "Waypoints",
    Options = {"Spawn", "VIP Room", "Arena"},
    Callback = function(opt)
        Window:Notify({Title = "Teleport", Content = "Moving to " .. opt, Duration = 3})
    end
})

Misc:CreateButton("Destroy UI", function()
    Window.ScreenGui:Destroy()
end)

Window:Notify({
    Title = "Ready",
    Content = "Script is fully loaded!",
    Duration = 5
})
